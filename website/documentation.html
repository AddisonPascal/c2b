# Documenting most aspects of c2b [Compiles to Batch] programming language.

# A comment's syntax is a hashtag [#] at the start of a line. Only one line comments are available in C2B.
# A comment

# To print a line to the console, use print[text]
print[This is a message. ]
print[Line 1 \n Line 2]
# Blank line:
print[]
# Inline colours and styles are available in print[], out[], and rewrite[], however compatibility is limited in non-Win10 operating systems.
# For more information, see Issue #1: https://github.com/AddisonPascal/c2b/issues/1
# Use @ to change colour, where is the hex colour code.
# Use # to change only text colour, not background.
# Use @bold to make it bold.
# Use @u or @underline to underline.
# Use @i or @inverse to inverse the colours.
# Use @r or @reset to reset to default.
# - Resets after every line.
# - Combinations are possible!
# Examples:
print[#a Green Text!]
print[@fa Green on White!]
print[@bold Bold]
print[@u underlined]
print[@fa@bold@u All 3!]

# Prints text to the console without a newline: out[text]
out[Hello, ]
out[World!]

# Rewrites the previous line: rewrite[text]
rewrite[Goodbye World]

# Changes the colour of the prompt: color[hex] or colour[hex]
color[0a]
colour[0c]

# Imports another module: import moduleName
# or: import net:moduleName
# or: import inbuilt:moduleName
# or: import local:moduleName
import test
import net:test
import inbuilt:test
import local:test

# Raise an exception: raise Exception
raise ExampleException

# Waits a certain amount of time: wait[seconds]
wait[3]
# Waits for the user to press any key:
wait[user]

# Restarts the program: restart
restart

# Exits file:
end[file]

# Maximises screen: disp[max]
disp[max]

# Clears the screen: clear
clear

# If statement: if[statement] {
# content
# }
if[%this%==%that%] {
print[This is that]
}
# and operators are "&" which can be used in any conditional statement:
if[%a%==%b% & %c%==%d% & %e%==%f%] {
print[AND it works!]
}

# Tests if an event is on: if.event[eventName] {
# content
# }
if.event[testEvent] {
print[Event is on]
}

# Executes a Windows Batch / CMD command: cmd[command]
cmd[echo Hi]

# Executes a Windows PowerShell command: ps[command]
ps[echo("Hi!");]
ps[[console]::Beep();]

# Changes title: title[text]
title[This is the title. ]

# Makes a function: define.function[name] {
# content
# }
define.function[function] {
print[This is a function]
print[%arg1%]
print[%arg2%]
print[%arg3%]
}

# A place in the code which can be returned to later on with goto[]: place[name]
place[place]

# Goes to a specified place[] in the code: goto[place]
goto[place]

# Executes a function: $function arg1 arg2 arg3...
$function "Argument 1" "Argument 2" "Argument 3"

# Defines a variable: define[var=text] OR define.text[var=text]
# define.math[var=expression]
# define.prompt[var=prompt]
# [var=content]
# [var=expression]
define[textVar=This is text]
define.text[textVar2=This is also text]
define.math[mathVar=1+1]
define.prompt[userVar=Enter text here: ]
[textVar3=This is text]

# Saves input to variable: prompt[var]
prompt[userVar2]

# Saves the character [from a list] the user entered without pressing enter to variable %choice%: choice[range]
choice[abc]
print[You entered %choice%.]

# Downloads a file from the internet (saved as 'download'): download[URL]
download[http://raw.githubusercontent.com/AddisonPascal/c2b/master/C2B%20Compiler.bat]

# Renames a file: ren[originalname newname]
ren[filename newfilename]

# Deletes a file: del[file]
del[file]

# Makes a new file: mkfile[name]
mkfile[file]

# Makes a new folder: mkfolder[name]
mkfolder[folder]

# Plays a note: play[frequency(hz), time(ms)]
play[1000,1000]

# Writes the output of commands to a file: file.write[filename] {
# content
# }
file.write[file.txt] {
print[This will be in a file]
}

# Writes the output of commands to the end of a file: file.append[filename] {
# content
# }
file.append[file.txt] {
print[This will be at the bottom of a file]
}

# Executes commands while a condition is true: while[condition] {
# content
# }
define[a=0]
while[%a%<10] {
print[%a%]
incr[a]
}

# Increments a variable:
incr[var]

# Repeats commands: repeat[number]:
# content
# break[]
repeat[10] {
print[Hello World!]
}

# Breaks out of a loop: break
break

# Returns from a function: return <value&rt
return 10
# Functions can be called with the setf, mathf, printf, f$, and returnf statements:
# Note that the syntax is similar to a variable, but with $ instead of %:
setf[result=$function args$]
mathf[result=3+$function args$]
printf[Result: $function args$]
f$run $function args$
returnf $function args$
# However, recursion is not yet possible.

# Sends keys to the focused window: skey[keys]
skey[Hello{ENTER}World{!}]

# Opens a file, application, or website: open[filename/address]
open[file.txt]
open[notepad]
open[http://google.com]

# Stops a process or application: stop[app.exe] or close[app.exe]
stop[notepad.exe]
close[cmd.exe]

# Else statement: if the previous if statement wasn't true: if[condition] {
# content
# } else {
# content
# }
if[%this%==%that%] {
print[This is that.]
} else {
print[This is not that]
}

# Elif statement: if the previous if statement wasn't true, execute another if statement: if[condition] {
# content
# } elif[condition] {
# content
# }
if[%this%==%that%] {
print[This is that.]
} elif[%this%>%that%] {
print[This is not that, but this is better than that.]
}

# Alert: pops up a message box: alert[text]
alert[Hello!]

# Creates a new process: process.start[processName] {
# content
# }
process.start[testProcess] {
while[true] {
if.event[testEvent] {
event.stop[testEvent]
mkfolder[testFolder]
}
}
}

# Stops a process: process.stop[processName]
process.stop[testProcess]

# Starts an event: event.start[eventName]
event.start[testEvent]

# Stops an event: event.stop[eventName]
event.stop[testEvent]

# Waits for an event to start: event.wait[eventName]
event.wait[eventName]

# Ends compilation. All text past here is ignored: export
export
